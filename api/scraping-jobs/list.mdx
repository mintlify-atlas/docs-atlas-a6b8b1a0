## Overview

Retrieve scraping jobs with various filtering options. The library provides multiple functions to query jobs by different criteria.

## Get All Jobs

### Function Signature

```typescript
export async function getAllJobs(): Promise<ScrapingJob[]>
```

Retrieve all scraping jobs, ordered by creation date (most recent first).

### Response

Returns an array of `ScrapingJob` objects. Returns an empty array if no jobs exist or if an error occurs.

<Expandable title="ScrapingJob fields">
  <ResponseField name="job_id" type="string">
    Unique identifier for the job (UUID)
  </ResponseField>
  
  <ResponseField name="influencer_name" type="string">
    Name of the influencer
  </ResponseField>
  
  <ResponseField name="platform" type="Platform">
    Social media platform (instagram, threads, tiktok, or x)
  </ResponseField>
  
  <ResponseField name="airtable_base_id" type="string">
    Airtable base ID for data isolation
  </ResponseField>
  
  <ResponseField name="base_id" type="string">
    Multi-tenant isolation key
  </ResponseField>
  
  <ResponseField name="num_vas" type="number | null">
    Number of VAs assigned to this job
  </ResponseField>
  
  <ResponseField name="status" type="JobStatus">
    Current job status (active, paused, or archived)
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    ISO 8601 timestamp of job creation
  </ResponseField>
  
  <ResponseField name="updated_at" type="string | null">
    ISO 8601 timestamp of last update
  </ResponseField>
</Expandable>

### Example

```typescript
import { getAllJobs } from '@/lib/scraping-jobs';

const jobs = await getAllJobs();
console.log(`Found ${jobs.length} jobs`);

jobs.forEach(job => {
  console.log(`${job.influencer_name} - ${job.platform} - ${job.status}`);
});
```

## Get Jobs by Platform

### Function Signature

```typescript
export async function getJobsByPlatform(
  platform: Platform,
  activeOnly = false
): Promise<ScrapingJob[]>
```

Retrieve jobs for a specific social media platform.

### Parameters

<ParamField path="platform" type="Platform" required>
  The platform to filter by:
  - `instagram`
  - `threads`
  - `tiktok`
  - `x`
</ParamField>

<ParamField path="activeOnly" type="boolean" default="false">
  If `true`, only returns jobs with status='active'
</ParamField>

### Response

Returns an array of `ScrapingJob` objects matching the platform criteria.

### Example

<CodeGroup>
```typescript All Instagram Jobs
import { getJobsByPlatform } from '@/lib/scraping-jobs';

const instagramJobs = await getJobsByPlatform('instagram');
console.log(`Instagram jobs: ${instagramJobs.length}`);
```

```typescript Active TikTok Jobs Only
import { getJobsByPlatform } from '@/lib/scraping-jobs';

const activeTikTokJobs = await getJobsByPlatform('tiktok', true);
console.log(`Active TikTok jobs: ${activeTikTokJobs.length}`);
```

```typescript React Component
import { getJobsByPlatform } from '@/lib/scraping-jobs';
import { useEffect, useState } from 'react';

function PlatformJobList({ platform }: { platform: Platform }) {
  const [jobs, setJobs] = useState<ScrapingJob[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function fetchJobs() {
      const jobsData = await getJobsByPlatform(platform);
      setJobs(jobsData);
      setLoading(false);
    }
    fetchJobs();
  }, [platform]);

  if (loading) return <div>Loading...</div>;

  return (
    <div>
      <h2>{platform} Jobs ({jobs.length})</h2>
      {jobs.map(job => (
        <JobCard key={job.job_id} job={job} />
      ))}
    </div>
  );
}
```
</CodeGroup>

## Get Jobs by Status

### Function Signature

```typescript
export async function getJobsByStatus(
  status: JobStatus
): Promise<ScrapingJob[]>
```

Retrieve jobs with a specific status.

### Parameters

<ParamField path="status" type="JobStatus" required>
  The status to filter by:
  - `active` - Currently running jobs
  - `paused` - Temporarily paused jobs
  - `archived` - Archived jobs
</ParamField>

### Response

Returns an array of `ScrapingJob` objects with the specified status.

### Example

```typescript
import { getJobsByStatus } from '@/lib/scraping-jobs';

// Get all active jobs
const activeJobs = await getJobsByStatus('active');
console.log(`Active jobs: ${activeJobs.length}`);

// Get paused jobs for review
const pausedJobs = await getJobsByStatus('paused');
console.log(`Paused jobs: ${pausedJobs.length}`);
```

## Get Single Job by ID

### Function Signature

```typescript
export async function getJobById(
  jobId: string
): Promise<ScrapingJob | null>
```

Retrieve a specific job by its ID.

### Parameters

<ParamField path="jobId" type="string" required>
  The UUID of the job to retrieve
</ParamField>

### Response

Returns a `ScrapingJob` object if found, or `null` if not found or error occurs.

### Example

```typescript
import { getJobById } from '@/lib/scraping-jobs';

const job = await getJobById('123e4567-e89b-12d3-a456-426614174000');

if (job) {
  console.log(`Found job for ${job.influencer_name}`);
} else {
  console.log('Job not found');
}
```

## Search Jobs by Influencer

### Function Signature

```typescript
export async function searchJobsByInfluencer(
  searchTerm: string
): Promise<ScrapingJob[]>
```

Search for jobs by influencer name using case-insensitive pattern matching.

### Parameters

<ParamField path="searchTerm" type="string" required>
  The search term to match against influencer names (case-insensitive)
</ParamField>

### Response

Returns an array of `ScrapingJob` objects where the influencer name contains the search term.

### Example

```typescript
import { searchJobsByInfluencer } from '@/lib/scraping-jobs';

// Search for jobs containing "john"
const results = await searchJobsByInfluencer('john');
console.log(`Found ${results.length} jobs`);

// Example: "John Doe", "Johnny Smith", etc.
results.forEach(job => {
  console.log(job.influencer_name);
});
```

## Get Jobs with Statistics

### Function Signature

```typescript
export async function getAllJobsWithStats(): Promise<ScrapingJobWithStats[]>
```

Retrieve all jobs with their associated statistics (usernames, assignments, scrape results).

### Response

Returns an array of `ScrapingJobWithStats` objects.

<Expandable title="ScrapingJobWithStats fields">
  <ResponseField name="...ScrapingJob" type="ScrapingJob">
    All standard ScrapingJob fields
  </ResponseField>
  
  <ResponseField name="stats" type="JobStatistics">
    Aggregated statistics for the job
    
    <Expandable title="JobStatistics fields">
      <ResponseField name="totalUsernames" type="number">
        Total number of usernames for this job
      </ResponseField>
      
      <ResponseField name="usedUsernames" type="number">
        Number of usernames that have been used
      </ResponseField>
      
      <ResponseField name="availableUsernames" type="number">
        Number of usernames still available
      </ResponseField>
      
      <ResponseField name="totalAssignments" type="number">
        Total number of daily assignments
      </ResponseField>
      
      <ResponseField name="totalScrapeResults" type="number">
        Total number of scrape results
      </ResponseField>
    </Expandable>
  </ResponseField>
</Expandable>

### Example

<CodeGroup>
```typescript Basic Usage
import { getAllJobsWithStats } from '@/lib/scraping-jobs';

const jobsWithStats = await getAllJobsWithStats();

jobsWithStats.forEach(job => {
  console.log(`${job.influencer_name}:`);
  console.log(`  Total usernames: ${job.stats.totalUsernames}`);
  console.log(`  Available: ${job.stats.availableUsernames}`);
  console.log(`  Assignments: ${job.stats.totalAssignments}`);
});
```

```typescript React Dashboard
import { getAllJobsWithStats } from '@/lib/scraping-jobs';
import { useEffect, useState } from 'react';

function JobsDashboard() {
  const [jobs, setJobs] = useState<ScrapingJobWithStats[]>([]);

  useEffect(() => {
    async function loadJobs() {
      const data = await getAllJobsWithStats();
      setJobs(data);
    }
    loadJobs();
  }, []);

  return (
    <div>
      {jobs.map(job => (
        <div key={job.job_id}>
          <h3>{job.influencer_name}</h3>
          <p>Platform: {job.platform}</p>
          <p>Status: {job.status}</p>
          <div className="stats">
            <span>Usernames: {job.stats.totalUsernames}</span>
            <span>Available: {job.stats.availableUsernames}</span>
            <span>Used: {job.stats.usedUsernames}</span>
          </div>
        </div>
      ))}
    </div>
  );
}
```
</CodeGroup>

## Error Handling

<Note>
All list functions return empty arrays instead of throwing errors. Check the console for detailed error messages.
</Note>

```typescript
const jobs = await getAllJobs();

if (jobs.length === 0) {
  console.log('No jobs found or error occurred');
  // Check browser console for error details
}
```

## Type Definitions

```typescript
type Platform = 'instagram' | 'threads' | 'tiktok' | 'x';

type JobStatus = 'active' | 'paused' | 'archived';

interface ScrapingJob {
  job_id: string;
  influencer_name: string;
  platform: Platform;
  airtable_base_id: string;
  base_id: string;
  num_vas: number | null;
  status: JobStatus;
  created_at: string;
  updated_at: string | null;
}

interface JobStatistics {
  totalUsernames: number;
  usedUsernames: number;
  availableUsernames: number;
  totalAssignments: number;
  totalScrapeResults: number;
}

interface ScrapingJobWithStats extends ScrapingJob {
  stats: JobStatistics;
}
```
